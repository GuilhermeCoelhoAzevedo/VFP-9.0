select TOP 50

PRO.FK_CADUNICO,
MAX(CAD.DS_FANTASIA) AS DS_FANTASIA,
SUM(APO.NR_DURACAO/60) AS NR_DURACAO_HS,
CAST(ROUND( SUM(APO.NR_DURACAO/60/8), 0) AS INTEGER) AS NR_DURACAO_DIAS,
LEFT( CONVERT( CHAR(10), MIN(APO.DT_APONTAMENTO), 111 ), 7) AS DT_MINIMA,
LEFT( CONVERT( CHAR(10), MAX(APO.DT_APONTAMENTO), 111 ), 7) AS DT_MAXIMA

from os_apontamentos APO

LEFT JOIN OS_ORDENS ORD ON ORD.PK_ID = APO.FK_ORDEM
LEFT JOIN OS_PROJETOS PRO ON PRO.PK_ID = ORD.FK_PROJETO
LEFT JOIN TB_CADUNICO CAD ON CAD.PK_ID = PRO.FK_CADUNICO

where APO.dt_apontamento >= CONVERT(DateTime,'01/01/2012',103)
and APO.dt_apontamento <= CONVERT(DateTime,'31/12/2012',103)

GROUP BY PRO.FK_CADUNICO

ORDER BY NR_DURACAO_HS DESC
